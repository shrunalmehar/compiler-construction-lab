%token NUM PLUS MINUS MUL DIV

%%
line: expr '\n' { printf("Result = %d\n", $1); }
    | '\n'
    ;
expr: NUM                { $$ = $1; }
    | expr PLUS expr    { $$ = $1 + $3; }
    | expr MINUS expr   { $$ = $1 - $3; }
    | expr MUL expr     { $$ = $1 * $3; }
    | expr DIV expr     { $$ = $1 / $3; }
    ;
%%
int main(void){ printf("Enter expression and newline:\\n"); yyparse(); return 0; }
